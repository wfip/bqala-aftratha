function checkout() { if (cart.length === 0) { alert("السلة فارغة حالياً! قم بإضافة المنتجات أولاً."); return; } // حفظ الطلب في الذاكرة المشتركة لتراه لوحة التحكم let existingOrders = JSON.parse(localStorage.getItem('grocery_orders')) || []; let newOrder = { id: Math.floor(1000 + Math.random() * 9000), date: new Date().toLocaleString('ar-EG'), items: cart, total: document.getElementById('cart-total').innerText }; existingOrders.push(newOrder); localStorage.setItem('grocery_orders', JSON.stringify(existingOrders)); alert(`🎉 تم إرسال طلبك بنجاح إلى لوحة التحكم!\nرقم الطلب: #${newOrder.id}`); clearCart(); }
